Relay

Production-ready GraphQL client for React

Relay: Core Modules

  1. Relay Compiler
    Transforms & optimizes queries + support for custom directives
  2. Relay Runtime
    High-performance GraphQL runtime that can be used to build higher-level client APIs
  3. React/Relay
    Product API that integrates the Relay Runtime with React

React/Relay: Fetch

import { useLazyLoadQuery } from 'react-relay/hooks';
import type { EditProfileGraphQLQuery } from
  '../__generated__/EditProfileGraphQLQuery.graphql.js';

export default function EditProfileGraphQL() {
  const data = useLazyLoadQuery<EditProfileGraphQLQuery>(query);
  const { pinner } = data.me ?? {};

  return <Box>{pinner.firstName}</Box>;
}

React/Relay: Mutation

import type {SaveButtonMutation} from 'SaveButtonMutation.graphql';
const {useMutation} = require('react-relay/hooks');

export default function SaveButton(props: {| +pinId: string |}) {
  const [commit, isInFlight] = useMutation<SaveButtonMutation>(mutationQuery);

  const mutationConfig = {
    variables: {
      input: props.pinId,
    },
  };

  return (
    <Button disabled={isInFlight} onClick={() => commit(mutationConfig)}>
      Save
    </Button>
  );
}

React/Relay: Mutation

export default function SaveButton(props: {| +pinId: string |}) {
  const [commit, isInFlight] = useMutation<SaveButtonMutation>(mutationQuery);

  const mutationConfig = {
    onCompleted: (data) => {
      console.log(data);
    },
    variables: {
      input: props.pinId,
    },
  };

  return (
    <Button disabled={isInFlight} onClick={() => commit(mutationConfig)}>
      Save
    </Button>
  );
}
